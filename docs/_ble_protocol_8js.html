◆ buildSysSettingCmd

export const buildSysSettingCmd
Initial value:
= (mode, freq) => {
const buffer = Buffer.alloc(3);
buffer.writeUInt8(COMMANDS.CMD_SYS_SETTING_SET, 0);
buffer.writeUInt8(mode, 1);
buffer.writeUInt8(freq, 2);
return buffer.toString('base64');
}
