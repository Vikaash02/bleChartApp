34 const connectAndStart = async (scannedDevice) => {

53 setStatus('Verifying Parameters...');

55 await connectedDevice.writeCharacteristicWithResponseForService(

58 readPayload

138 // Keep only last 100 points for chart performance

141 // Clear buffer
